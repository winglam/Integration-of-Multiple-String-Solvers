            f: b

          ];

# L(c2) = rrs

            f: f
            d: c -> d on { 'r' }
            d -> e on { 'r' }
	    e -> f on { 's' }
          ];

# L(c3) = qr*s
c3 < [ s: g
            f: i
            d: g -> h on { 'q' }
            h -> h on { 'r' }
            h -> i on { 's' }];

# qr*rrs intersect qr*s
c1 . c2 < c3;
